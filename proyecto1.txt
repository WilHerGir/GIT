function calcularTriangulo(base, altura) {

    let area = (base * altura) / 2;
    let perimetro = base + altura + Math.sqrt(base ** 2 + altura ** 2)

    console.log(`Área del triángulo: ${area}`);
    console.log(`Perímetro del triángulo: ${perimetro}`);
}
